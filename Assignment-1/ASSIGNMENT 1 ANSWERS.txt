Topic 1: Structures

#include <stdio.h>
#include <string.h>

#define DAYS 7
#define MAX_TASKS 3
#define MAX_LEN 50

struct Day {
    char dayName[15];
    char tasks[MAX_TASKS][MAX_LEN];
    int taskCount;
};

int main() {
    struct Day week[DAYS] = {
        {"Monday", {}, 0},
        {"Tuesday", {}, 0},
        {"Wednesday", {}, 0},
        {"Thursday", {}, 0},
        {"Friday", {}, 0},
        {"Saturday", {}, 0},
        {"Sunday", {}, 0}
    };

    int dayIndex, numTasks;

    printf("Enter day number (0-Monday ... 6-Sunday): ");
    scanf("%d", &dayIndex);

    if(dayIndex < 0 || dayIndex > 6) {
        printf("Invalid day.\n");
        return 1;
    }

    printf("Enter number of tasks (max 3): ");
    scanf("%d", &numTasks);

    getchar();

    if(numTasks > MAX_TASKS)
        numTasks = MAX_TASKS;

    week[dayIndex].taskCount = numTasks;

    for(int i = 0; i < numTasks; i++) {
        printf("Enter task %d: ", i + 1);
        fgets(week[dayIndex].tasks[i], MAX_LEN, stdin);
        week[dayIndex].tasks[i][strcspn(week[dayIndex].tasks[i], "\n")] = '\0';
    }

    printf("\nWeekly Calendar\n");

    for(int i = 0; i < DAYS; i++) {
        printf("\n%s:\n", week[i].dayName);

        if(week[i].taskCount == 0) {
            printf("No Tasks\n");
        }
        else {
            for(int j = 0; j < week[i].taskCount; j++) {
                printf("- %s\n", week[i].tasks[j]);
            }
        }
    }

    return 0;
}

Topic 2: Pointers

#include <stdio.h>
#include <stdlib.h>

void rearrange(int *arr, int size)
{
    int *temp = (int*)malloc(size * sizeof(int));

    int *ptr = temp;

    for(int *p = arr; p < arr + size; p++)
    {
        if((*p) % 2 == 0)
        {
            *ptr = *p;
            ptr++;
        }
    }

    for(int *p = arr; p < arr + size; p++)
    {
        if((*p) % 2 != 0)
        {
            *ptr = *p;
            ptr++;
        }
    }

    int *t = temp;

    for(int *p = arr; p < arr + size; p++)
    {
        *p = *t;
        t++;
    }

    free(temp);
}

int main()
{
    int arr[] = {7, 2, 9, 4, 6, 3, 8, 1};
    int size = sizeof(arr) / sizeof(arr[0]);

    printf("Original Array:\n");

    for(int *p = arr; p < arr + size; p++)
        printf("%d ", *p);

    rearrange(arr, size);

    printf("\n\nRearranged Array:\n");

    for(int *p = arr; p < arr + size; p++)
        printf("%d ", *p);

    return 0;
}

Topic 3: Arrays

#include <stdio.h>
#include <stdbool.h>

bool searchMatrix(int matrix[][4], int n, int key)
{
    int row = 0;
    int col = n - 1;

    while(row < n && col >= 0)
    {
        if(matrix[row][col] == key)
            return true;

        if(matrix[row][col] > key)
            col--;
        else
            row++;
    }

    return false;
}

int main()
{
    int matrix[4][4] = {
        {10, 20, 30, 40},
        {15, 25, 35, 45},
        {27, 29, 37, 48},
        {32, 33, 39, 50}
    };

    int key = 29;

    if(searchMatrix(matrix, 4, key))
        printf("Key Found\n");
    else
        printf("Key Not Found\n");

    return 0;
}

Sample Output:
Key Found

Time Complexity:
O(n)

Space Complexity
O(1)

This is the most efficient approach because it starts from the top-right corner and eliminates either a row or a column in each step.
